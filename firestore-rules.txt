// Recommended Firestore Security Rules for HifzMate
// Copy and paste these into Firebase Console > Firestore Database > Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // User data - users can only read/write their own data
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Optional: If you add other collections later, add rules here
    // Example for a public collection (read-only for all authenticated users):
    // match /public/{document=**} {
    //   allow read: if request.auth != null;
    //   allow write: if false; // No one can write
    // }
    
    // Default: Deny all other access
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
